17 inline void baseOp_maxpool(CLTensor &tensor,
18 const uint32_t pool_size,
19 const uint32_t stride,
20 const bool ceil_mode,
21 const OpMemAllocator &memAllocator,
22 const bool releaseInputData = false) {
23 if (!tensor.data) {
24 return;
25 }
26 cl_mem ori_data = tensor.data;
27 uint32_t output_M = floor((float) (tensor.M - pool_size) / (float) stride + 1);
28 uint32_t output_N = floor((float) (tensor.N - pool_size) / (float) stride + 1);
29 //support ceil mode.
30 if (ceil_mode) {
31 output_M = ceil((float) (tensor.M - pool_size) / (float) stride + 1);
32 output_N = ceil((float) (tensor.N - pool_size) / (float) stride + 1);
33 }
34 uint32_t output_channel = tensor.channel_in;
35 auto output_data_type = tensor.data_type;
36 cl_mem output_buffer = memAllocator(output_data_type.getBytes()
37 * output_M
38 * output_N
39 * output_channel);
40
41 tensor.data = clFrame.run_S_maxpooling(tensor.data,
42 output_buffer,
43 output_data_type,
44 tensor.M,
45 tensor.N,
46 tensor.channel_in,
47 pool_size,
48 stride,
49 ceil_mode);
50
51 if (tensor.data) {
52 tensor.M = output_M;
53 tensor.N = output_N;
54 tensor.MNL = tensor.M * tensor.N * tensor.channel_in;
55 tensor.data_type = output_data_type;
56 tensor.bytes = tensor.MNL * tensor.data_type.getBytes();
57 }
58 if (releaseInputData) {
59 clFrame.releaseMemObj(ori_data);
60 }
61 }
