19 inline void baseOp_concat(std::vector<CLTensor *> &inputs,
20 CLTensor &out,
21 const uint32_t concat_direction,
22 const OpMemAllocator &memAllocator,
23 const bool releaseInputData = false) {
24 if (inputs.size() < 2) {
25 showError("Error at baseOp_concat:, invalid inputs size: " +
26 std::to_string(inputs.size()) + ", need two and more inputs.");
27 return;
28 }
29 uint32_t buffer_size = 0;
30 std::vector<cl_mem> il;
31 std::vector<uint32_t> ds;
32 std::vector<uint32_t> hs;
33 std::vector<uint32_t> ws;
34 std::vector<uint32_t> cs;
35 DataType dt = inputs[0]->data_type;
36 for (int i = 0; i < inputs.size(); ++i) {
37 if (inputs[i]->data_type != dt) {
38 showError("Error at baseOp_concat: "
39 "need inputs with same data type");
40 return;
41 }
42 buffer_size += inputs[i]->bytes;
43 il.push_back(inputs[i]->data);
44 ds.push_back(inputs[i]->channel_out);
45 hs.push_back(inputs[i]->M);
46 ws.push_back(inputs[i]->N);
47 cs.push_back(inputs[i]->channel_in);
48 }
49 cl_mem output_buffer = memAllocator(buffer_size);
50 if (output_buffer == nullptr)
51 return;
52 out.data = clFrame.run_S_concat(il, output_buffer, dt,
53 ds, hs, ws, cs, concat_direction);
54 if (out.data) {
55 out.channel_out = inputs[0]->channel_out;
56 out.M = inputs[0]->M;
57 out.N = inputs[0]->N;
58 out.channel_in = inputs[0]->channel_in;
59 out.channel_in_valid = inputs[0]->channel_in_valid;
60 switch (concat_direction) {
61 case CONCAT_DIR::D: {
62 out.channel_out = 0;
63 for (auto i : ds) {
64 out.channel_out += (uint32_t) i;
65 }
66 }
67 break;
68 case CONCAT_DIR::H: {
69 out.M = 0;
70 for (auto i : hs) {
71 out.M += (uint32_t) i;
72 }
73 }
74 break;
75 case CONCAT_DIR::W: {
76 out.N = 0;
77 for (auto i : ws) {
78 out.N += (uint32_t) i;
79 }
80 }
81 break;
82 case CONCAT_DIR::C: {
83 //NOTE: This function is NOT support BNN data type.
84 //Need to extend it.
85 out.channel_in = 0;
86 for (auto i : cs) {
87 out.channel_in += (uint32_t) i;
88 }
89 out.channel_in_valid = out.channel_in;
90 }
91 break;
92 default: {
93
94 }
95 break;
96 }
97 uint32_t a = out.channel_out * out.M * out.N * out.channel_in * out.data_type.getBytes();
98 if (buffer_size != a) {
99 showError("error at baseOp_concat: "
100 "output data size is not matched with CLTensor's dim sizes.");
101 out.clear();
102 }
103 out.bytes = buffer_size;
104 }
105 if (releaseInputData) {
106 for (auto &tensor: inputs) {
107 clFrame.releaseMemObj(tensor->data);
108 }
109 }
110 }
111
112}
