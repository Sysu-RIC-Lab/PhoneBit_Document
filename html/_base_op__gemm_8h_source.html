24 inline void baseOp_fgemm_bn_add(CLTensor &tensor,
25 CLTensor &tensor_packed,
26 const CLTensor &weight,
27 const CLTensor *const thr,
28 const CLTensor *const GDS,
29 const CLTensor *const add,
30 const OpGemmOutType outType,
31 const OpMemAllocator &memAllocator,
32 const bool releaseInputData = false) {
33 if (tensor.data_type != Float16 && tensor.data_type != Float32 && weight.data_type != Float16 &&
34 weight.data_type != Float32) {
35 showError("Error at baseOp_fgemm_bn_add: support tensor with Float16/Float32 data type only!");
36 return;
37 }
38 if (!weight.data) {
39 showError("Error at baseOp_fgemm_bn_add: weight's data is null!");
40 return;
41 }
42 // gemm methods only process a 2D tensor(both channel_out and channel_in are 1)
43 if (tensor.channel_in != 1 || tensor.channel_out != 1 || weight.channel_in != 1 || weight.channel_out != 1) {
44 showError("Error at baseOp_fgemm_bn_add: c and n in tensor and weight must be 1!");
45 return;
46 }
47 if (tensor.data) {
48 cl_mem originalData = tensor.data;
49// uint32_t output_M = tensor.M_after_conv;
50// uint32_t output_N = tensor.N_after_conv;
51 uint32_t output_M = tensor.M;
52 uint32_t output_N = weight.N;
53 uint32_t output_channel_in = 1;
54// uint32_t output_channel_in_valid = weight.channel_out_valid;
55 uint32_t output_channel_out = 1;//for now is 1
56// uint32_t output_channel_out_valid = output_channel_out;//for now is 1
57 DataType output_data_type_FType = Float32;
58 DataType output_data_type_Bit = Bit4;
59 if (clFrame.isFp16Enabled()) {
60 output_data_type_FType = Float16;
61 }
62 uint32_t output_FType_bytes = output_data_type_FType.getBytes()
63 * output_channel_out * output_M * output_N * output_channel_in;
64 uint32_t output_Bit_bytes = output_data_type_Bit.getBytes()
65 * output_channel_out * output_M * ((output_N + 3) / 4) *
66 output_channel_in;
67 cl_mem output_data_FType = nullptr;
68 cl_mem output_data_Bit = nullptr;
69 cl_mem thr_data = nullptr;
70 cl_mem GDS_data = nullptr;
71 cl_mem add_data = nullptr;
72 if (thr) {
73 thr_data = thr->data;
74 }
75 if (GDS) {
76 GDS_data = GDS->data;
77 }
78 if (add) {
79 add_data = add->data;
80 }
81 switch (outType) {
82 default:
83 case FLOAT: {
84 output_data_FType = memAllocator(output_FType_bytes);
85 output_data_Bit = nullptr;
86 }
87 break;
88 case BNN: {
89 output_data_FType = nullptr;
90 output_data_Bit = memAllocator(output_Bit_bytes);
91 }
92 break;
93 case FLOAT_BNN: {
94 output_data_FType = memAllocator(output_FType_bytes);
95 output_data_Bit = memAllocator(output_Bit_bytes);
96 }
97 break;
98 }
99 clFrame.run_S_fgemm_opt_4x4(tensor.data, output_data_FType, output_data_Bit,
100 weight.data, thr_data, GDS_data, add_data,
101 tensor.M, tensor.N, weight.N);
102 switch (outType) {
103 default:
104 case FLOAT: {
105 tensor_packed.clear();
106 tensor.M = output_M;
107 tensor.N = output_N;
108 tensor.channel_in = output_channel_in;
109// tensor.channel_in_valid = output_channel_in_valid;
110 tensor.channel_out = output_channel_out;
111 tensor.channel_out_valid = output_channel_out;
112 tensor.data = output_data_FType;
113 tensor.data_type = output_data_type_FType;
114 tensor.bytes = output_FType_bytes;
115 }
116 break;
117 case BNN: {
118 tensor.clear();
119 tensor_packed.M = output_M;
120 tensor_packed.N = (output_N + 3) / 4;
121 tensor_packed.channel_in = output_channel_in;
122// tensor_packed.channel_in_valid = output_channel_in_valid;
123 tensor_packed.channel_out = output_channel_out;
124// tensor_packed.channel_out_valid = output_channel_out_valid;
125 tensor_packed.data = output_data_Bit;
126 tensor_packed.data_type = output_data_type_Bit;
127 tensor_packed.bytes = output_Bit_bytes;
128 }
129 break;
130 case FLOAT_BNN: {
131 tensor.M = output_M;
132 tensor.N = output_N;
133 tensor.channel_in = output_channel_in;
134// tensor.channel_in_valid = output_channel_in_valid;
135 tensor.channel_out = output_channel_out;
136// tensor.channel_out_valid = output_channel_out_valid;
137 tensor.data = output_data_FType;
138 tensor.data_type = output_data_type_FType;
139 tensor.bytes = output_FType_bytes;
140
141 tensor_packed.M = output_M;
142 tensor_packed.N = (output_N + 3) / 4;
143 tensor_packed.channel_in = output_channel_in;
144// tensor_packed.channel_in_valid = output_channel_in_valid;
145 tensor_packed.channel_out = output_channel_out;
146// tensor_packed.channel_out_valid = output_channel_out_valid;
147 tensor_packed.data = output_data_Bit;
148 tensor_packed.data_type = output_data_type_Bit;
149 tensor_packed.bytes = output_Bit_bytes;
150 }
151 break;
152 }
153 if (releaseInputData) {
154 clFrame.releaseMemObj(originalData);
155 }
156 }
